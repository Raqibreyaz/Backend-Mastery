    <script>
        const image1Input = document.getElementById('image1');
        const image2Input = document.getElementById('image2');
        const label1 = document.getElementById('label1');
        const label2 = document.getElementById('label2');
        const preview1 = document.getElementById('preview1');
        const preview2 = document.getElementById('preview2');
        const form = document.getElementById('uploadForm');
        const statusMessage = document.getElementById('statusMessage');
        const submitBtn = document.getElementById('submitBtn');

        // Handle file input changes
        function handleFileSelect(input, label, preview) {
            input.addEventListener('change', function(e) {
                const file = e.target.files[0];
                if (file) {
                    label.classList.add('has-file');
                    label.innerHTML = `<span class="file-name">‚úì ${file.name}</span>`;

                    // Show preview
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        preview.classList.remove('empty');
                        preview.innerHTML = `<img src="${e.target.result}" alt="Preview">`;
                    };
                    reader.readAsDataURL(file);
                } else {
                    label.classList.remove('has-file');
                    label.innerHTML = '<span>üìÅ Click to select image</span>';
                    preview.classList.add('empty');
                    preview.innerHTML = preview.id === 'preview1' ? 'Preview 1' : 'Preview 2';
                }
            });
        }

        handleFileSelect(image1Input, label1, preview1);
        handleFileSelect(image2Input, label2, preview2);

        // Handle form submission
        form.addEventListener('submit', async function(e) {
            e.preventDefault();

            const image1 = image1Input.files[0];
            const image2 = image2Input.files[0];
            const prompt = document.getElementById('prompt').value;

            // Validate inputs
            if (!image1 || !image2) {
                showStatus('Please select both images', 'error');
                return;
            }

            if (!prompt.trim()) {
                showStatus('Please enter a prompt', 'error');
                return;
            }

            // Prepare form data
            const formData = new FormData();
            formData.append('image1', image1);
            formData.append('image2', image2);
            formData.append('prompt', prompt);

            // Disable submit button
            submitBtn.disabled = true;
            showStatus('Uploading...', 'loading');

            try {
                // Replace with your actual API endpoint
                const API_ENDPOINT = 'https://your-api-endpoint.com/upload';

                const response = await fetch(API_ENDPOINT, {
                    method: 'POST',
                    body: formData,
                    // Add headers if needed (e.g., authentication)
                    // headers: {
                    //     'Authorization': 'Bearer YOUR_TOKEN'
                    // }
                });

                if (response.ok) {
                    const result = await response.json();
                    showStatus('Successfully uploaded! ‚úì', 'success');
                    console.log('Response:', result);

                    // Reset form after 2 seconds
                    setTimeout(() => {
                        form.reset();
                        label1.classList.remove('has-file');
                        label2.classList.remove('has-file');
                        label1.innerHTML = '<span>üìÅ Click to select first image</span>';
                        label2.innerHTML = '<span>üìÅ Click to select second image</span>';
                        preview1.classList.add('empty');
                        preview2.classList.add('empty');
                        preview1.innerHTML = 'Preview 1';
                        preview2.innerHTML = 'Preview 2';
                        statusMessage.style.display = 'none';
                    }, 2000);
                } else {
                    const errorText = await response.text();
                    showStatus(`Upload failed: ${response.status} ${response.statusText}`, 'error');
                    console.error('Error:', errorText);
                }
            } catch (error) {
                showStatus(`Network error: ${error.message}`, 'error');
                console.error('Error:', error);
            } finally {
                submitBtn.disabled = false;
            }
        });

        function showStatus(message, type) {
            statusMessage.textContent = message;
            statusMessage.className = `status-message ${type}`;
            statusMessage.style.display = 'block';
        }
    </script>
